<?php
// Database connection variables
$servername = "localhost";
$username = "root"; // Default XAMPP username
$password = "";     // Default XAMPP password
$dbname = "inquiries"; // actual database name here

// Create connection
$conn = new mysqli($servername, $username, $password, $dbname);

// Check connection
if ($conn->connect_error) {
    die("Connection failed: " . $conn->connect_error);
}

// Check if form was submitted
if (isset($_POST['submit_enquiry'])) {
    
    // Collect and sanitize data
    $inst_name = mysqli_real_escape_string($conn, $_POST['institution_name']);
    $off_email = mysqli_real_escape_string($conn, $_POST['official_email']);
    $details   = mysqli_real_escape_string($conn, $_POST['enquiry_details']);

    // SQL query to insert data
    $sql = "INSERT INTO institution_inquiries (institution_name, official_email, enquiry_details) 
            VALUES ('$inst_name', '$off_email', '$details')";

    if ($conn->query($sql) === TRUE) {
        // Success: Alert the user and redirect back to the portfolio
        echo "<script>
                alert('Thank you! Your enquiry has been received.');
                window.location.href='index.html'; 
              </script>";
    } else {
        echo "Error: " . $sql . "<br>" . $conn->error;
    }
}

$conn->close();
?>
